--Create Document count
SELECT 
 G.PLC_LOC_CODE,PLC_LOCADESC, COUNT(*)
FROM HIL.GI_GU_DH_DOC_HEADER G
LEFT OUTER JOIN PR_GN_LC_LOCATION L ON (L.PLC_LOC_CODE=G.PLC_LOC_CODE)
Where
GDH_YEAR = '2018'
 AND PDT_DOCTYPE = 'A'
 AND PDP_DEPT_CODE = '13'
 AND GDH_POSTING_TAG = 'Y'
GROUP BY G.PLC_LOC_CODE,PLC_LOCADESC
ORDER BY COUNT(*) DESC
